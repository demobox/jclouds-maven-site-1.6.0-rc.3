23  /**
24   * The SSL Termination feature allows a load balancer user to terminate SSL traffic at the load balancer layer versus 
25   * at the web server layer. A user may choose to configure SSL Termination using a key and an SSL certificate or an 
26   * (Intermediate) SSL certificate.
27   * <p/> 
28   * When SSL Termination is configured on a load balancer, a secure shadow server is created that listens only for 
29   * secure traffic on a user-specified port. This shadow server is only visible to and manageable by the system. 
30   * Existing or updated attributes on a load balancer with SSL Termination will also apply to its shadow server. 
31   * For example, if Connection Logging is enabled on an SSL load balancer, it will also be enabled on the shadow server 
32   * and Cloud Files logs will contain log files for both.
33   * <p/>
34   * Notes
35   * <ol>
36   * <li>SSL Termination may only be configured on load balancers with non-secure protocols. For example, SSL Termination 
37   * can be applied to an HTTP load balancer, but not to an HTTPS load balancer.</li>
38   * <li>SSL-terminated load balancers decrypt the traffic at the traffic manager and pass unencrypted traffic to the 
39   * back-end node. Because of this, the customer's back-end nodes don't know what protocol the client requested. 
40   * Therefore the X-Forwarded-Proto (XFP) header has been added for identifying the originating protocol of an HTTP 
41   * request as "http" or "https" depending on what protocol the client requested.</li>
42   * <li>Not every service will return certificates in the proper order. Please verify that your chain of certificates 
43   * matches that of walking up the chain from the domain to the CA root.</li>
44   * </ol>
45   * 
46   * Warning
47   * <ol>
48   * <li>If SSL is enabled on a load balancer that is configured with nodes that are NOT in the same datacenter, then 
49   * decrypted traffic will be sent in clear text over the public internet to the external node(s) and will no longer 
50   * be secure.</li>
51   * </ol>
52   *
53   * @see SSLTerminationAsyncApi
54   * @author Everett Toews
55   */
56  public interface SSLTerminationApi {
57     /**
58      * Create or update SSL termination.
59      * </p>
60      * The following attributes may be updated without overwriting a load balancer's existing SSL certificate and key 
61      * specifications:
62      * <ul>
63      * <li>enabled</li>
64      * <li>secureTrafficOnly</li>
65      * <li>securePort</li>
66      * </ul>
67      * These configurable attributes must be provided (individually or grouped) in a request without specifying any 
68      * certificate/key combination if a user does not want the system to overwrite the existing SSL certificate/key 
69      * configuration.
70      * </p>
71      * If a user wants to replace the existing SSL configuration, a new certificate, privatekey, and securePort 
72      * combination must be provided instead of, or in addition to, the optional/editable attributes.
73      */
74     void createOrUpdate(SSLTermination sslTermination);
75  
76     /**
77      * Get SSL termination info.
78      */
79     SSLTermination get();
80     
81     /**
82      * Delete SSL termination.
83      * 
84      * @return true on a successful delete, false if the SSL termination was not found
85      */
86     boolean delete();
87  }
