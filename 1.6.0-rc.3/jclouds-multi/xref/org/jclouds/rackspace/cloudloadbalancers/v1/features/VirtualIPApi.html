24  /**
25   * A virtual IP makes a load balancer accessible by clients. The load balancing service supports either a public VIP,
26   * routable on the public Internet, or a ServiceNet address, routable only within the region in which the load balancer

28   * <p/>
29   * 
30   * @see VirtualIPAsyncApi
31   * @author Everett Toews
32   */
33  public interface VirtualIPApi {
34     /**
35      * Create a new virtual IP.
36      */
37     VirtualIPWithId create(VirtualIP virtualIP);
38  
39     /**
40      * List the virtual IPs.
41      */
42     Iterable<VirtualIPWithId> list();
43     
44     /**
45      * Delete a virtual IP.
46      * 
47      * @see VirtualIPApi#delete(Iterable)
48      * 
49      * @return true on a successful delete, false if the virtual IP was not found
50      */
51     boolean delete(int id);
52     
53     /**
54      * Batch delete virtual IPs given the specified ids.
55      * 
56      * All load balancers must have at least one virtual IP associated with them at all times. Attempting to delete the
57      * last virtual IP will result in an exception. The current default limit is ten ids per request. Any 
58      * and all configuration data is immediately purged and is not recoverable. If one or more of the items in the list 
59      * cannot be removed due to its current status, an exception is thrown along with the ids of the ones the 
60      * system identified as potential failures for this request.
61      * 
62      * @return true on a successful delete, false if the virtual IP was not found
63      */
64     boolean delete(Iterable<Integer> ids);
